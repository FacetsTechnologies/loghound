 93 /**
 94  * Object array defining all the log level objects and their specific attributes.
 95  * @namespace Manager for LogHoundLevel objects.
 96  */
 97 var LogHoundLevels = [];
 98 /**
 99  * @param {String} name The name of the log level to retrieve.
100  * @returns {LogHoundLevel} The retrieved log level, or <code>null</code> if
101  * no match for the argumented name was found.
102  */
103 LogHoundLevels.getByName = function(name) {
104     name = name.toLowerCase();
105     for(var idx=0; idx<this.length; idx++) {
106         if(this[idx].getName() == name) {
107             return this[idx];
108         }
109     }
110     return null;
111 };
112 /**
113  * @param {integer} id The ID of the log level to retrieve.
114  * @returns {LogHoundLevel} The retrieved log level, or <code>null</code> if
115  * no match for the argumented ID was found.
116  */
117 LogHoundLevels.getById = function(id) {
118     for(var idx=0; idx<this.length; idx++) {
119         if(this[idx].getId() == id) {
120             return this[idx];
121         }
122     }
123     return null;
124 };
125 /**
126  * @param {Number|String} level The ID or name of the log level to retrieve.
127  * @returns {LogHoundLevel} The retrieved log level, or <code>null</code> if
128  * no match for the argumented level indicator was found.
129  */
130 LogHoundLevels.getLevel = function(level) {
131     if(level==null) { return null; }
132     if((typeof level)=='number') {
133         return this.getById(level);
134     } else if((level instanceof String) || (typeof level)=='string') {
135         return this.getByName(level);
136     } else if(!(level instanceof LogHoundLevel)) {
137         return null;
138     }
139     return level;
140 };
141 /**
142  * Takes a new log level definition and extends LogHoundLevel with it, then
143  * adds it to the LogHoundLevels array.
144  * @private
145  */
146 LogHoundLevels.addLevel = function(newLevelFn) {
147     FctsTools.extend(newLevelFn, LogHoundLevel);
148     var newLevel = new newLevelFn();
149     LogHoundLevels[newLevel.getName().toUpperCase()] = newLevel;
150     LogHoundLevels.push(newLevel);
151 };

1191 /**
1192  * Parses the Javascript "arguments" object and separates the atomic Log Hound
1193  * argument types.
1194  * @private
1195  */
1196 LogHound.prototype.parseLoggingArgs = function() {
1197     var argArray = [];
1198     for(var i=0; i<arguments.length; i++) {
1199         argArray[i] = arguments[i];
1200     }
1201     var msgRec = [];
1202     for(var i=0; i<argArray.length; i++) {
1203         if(argArray[i]==null) { continue; }
1204         if((typeof argArray[i])=='object' && !(argArray[i] instanceof Array) && argArray[i].length) {
1205             for(var j=0; j<argArray[i].length; j++) {
1206                 argArray[argArray.length] = argArray[i][j];
1207             }
1208         } else if(argArray[i] instanceof Array) {
1209             msgRec['tags'] = argArray[i];
1210             var tmpArray = [];
1211             for(var x=0; x<msgRec['tags'].length; x++) {
1212                 if(!FctsTools.isBlank(msgRec['tags'][x])) {
1213                     tmpArray[tmpArray.length] = msgRec['tags'][x];
1214                 }
1215             }
1216             msgRec['tags'] = tmpArray;
1217         } else if(argArray[i] instanceof LogHoundLevel) {
1218             msgRec['level'] = argArray[i];
1219         } else if(argArray[i] instanceof String || (typeof argArray[i])==='string') {
1220             msgRec['text'] = argArray[i];
1221         } else if(argArray[i] instanceof Error) {
1222             msgRec['error'] = argArray[i];
1223         }
1224     }
1225     return msgRec;
1226 };
